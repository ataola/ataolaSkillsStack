```js
function removeWithoutCopy(arr, item) {
    for(let i = 0; i < arr.length; i++) {
        if(arr[i] === item) {

        }
```

```js
function removeWithoutCopy(arr, item){
    var count=0;

        if(arr[i]!=item){
            arr[count++]=arr[i];
        }
    }
```
